users

bets
- id INT AUTO_INCREMENT PRIMARY KEY,
- user_id INT NOT NULL,
- date DATE NOT NULL,
- bet_type VARCHAR(50) NOT NULL,
- amount_spent DECIMAL(10,2) NOT NULL,
- amount_earned DECIMAL(10,2) NOT NULL,
- FOREIGN KEY (user_id) REFERENCES users(id)
